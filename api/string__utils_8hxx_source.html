62 template <class CharT, class SepT, class OutIt>
63 auto split_on_any_of(const std::basic_string<CharT>& s, const SepT& sep,
64  OutIt out)
65 {
66  using size_type = typename std::basic_string<CharT>::size_type;
67  size_type i1 = 0;
68  size_type i2;
69  do {
70  i2 = s.find_first_of(sep, i1);
71  *out++ = s.substr(i1, i2 - i1);
72  i1 = i2 + 1; // we can only add +1 if sep is single char.
73 
74  // i2 gets s.npos after the last separator.
75  // Length of i2-i1 will always go past the end. That is defined.
76  } while (i2 != s.npos);
77  return out;
78 }

180 template <class CharT, class OutIt>
181 auto parse_on_whitespace(const std::basic_string<CharT>& s, OutIt out,
182  const std::locale& loc = std::locale()) -> OutIt
183 {
184  auto& f = std::use_facet<std::ctype<CharT>>(loc);
185  auto isspace = [&](auto& c) { return f.is(std::ctype_base::space, c); };
186  auto i1 = begin(s);
187  auto endd = end(s);
188  do {
189  auto i2 = std::find_if_not(i1, endd, isspace);
190  if (i2 != i1)
191  *out++ = std::basic_string<CharT>(i1, i2);
192  if (i2 == endd)
193  break;
194  auto i3 = std::find_if(i2, endd, isspace);
195  *out++ = std::basic_string<CharT>(i2, i3);
196  i1 = i3;
197  } while (i1 != endd);
198  return out;
199 }

289 template <class CharT>
290 auto is_number(const std::basic_string<CharT>& s) -> bool
291 {
292  if (s.empty())
293  return false;
294 
295  auto it = begin(s);
296  if (s[0] == '-')
297  ++it;
298  while (it != end(s)) {
299  auto next = find_if(it, end(s),
300  [](auto c) { return c < '0' || c > '9'; });
301  if (next == it)
302  return false;
303  if (next == end(s))
304  return true;
305  it = next;
306  auto c = *it;
307  if (c == '.' || c == ',' || c == '-')
308  ++it;
309  else
310  return false;
311  }
312  return false;
313 }

315 template <class DataIter, class PatternIter, class FuncEq = std::equal_to<>>
316 auto match_simple_regex(DataIter data_first, DataIter data_last,
317  PatternIter pat_first, PatternIter pat_last,
318  FuncEq eq = FuncEq())
319 {
320  auto s = std::stack<std::pair<DataIter, PatternIter>>();
321  s.emplace(data_first, pat_first);
322  auto data_it = DataIter();
323  auto pat_it = PatternIter();
324  while (!s.empty()) {
325  std::tie(data_it, pat_it) = s.top();
326  s.pop();
327  if (pat_it == pat_last) {
328  if (data_it == data_last)
329  return true;
330  else
331  return false;
332  }
333  auto node_type = *pat_it;
334  if (pat_it + 1 == pat_last)
335  node_type = 0;
336  else
337  node_type = *(pat_it + 1);
338  switch (node_type) {
339  case '?':
340  s.emplace(data_it, pat_it + 2);
341  if (data_it != data_last && eq(*data_it, *pat_it))
342  s.emplace(data_it + 1, pat_it + 2);
343  break;
344  case '*':
345  s.emplace(data_it, pat_it + 2);
346  if (data_it != data_last && eq(*data_it, *pat_it))
347  s.emplace(data_it + 1, pat_it);
348 
349  break;
350  default:
351  if (data_it != data_last && eq(*data_it, *pat_it))
352  s.emplace(data_it + 1, pat_it + 1);
353  break;
354  }
355  }
356  return false;
357 }
