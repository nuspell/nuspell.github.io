429 template <class Value, class Key = Value, class KeyExtract = identity,
430  class Hash = std::hash<Key>, class KeyEqual = std::equal_to<Key>>
431 class Hash_Multiset {
432  private:
433  using bucket_type = boost::container::small_vector<Value, 1>;
434  static constexpr float max_load_fact = 7.0 / 8.0;
435  std::vector<bucket_type> data;
436  size_t sz = 0;
437  size_t max_load_factor_capacity = 0;
438  KeyExtract key_extract = {};
439  Hash hash = {};
440  KeyEqual equal = {};
441 
442  public:
443  using key_type = Key;
444  using value_type = Value;
445  using size_type = std::size_t;
446  using difference_type = std::ptrdiff_t;
447  using hasher = Hash;
448  using reference = value_type&;
449  using const_reference = const value_type&;
450  using pointer = typename bucket_type::pointer;
451  using const_pointer = typename bucket_type::const_pointer;
452  using local_iterator = typename bucket_type::iterator;
453  using local_const_iterator = typename bucket_type::const_iterator;
454 
455  Hash_Multiset() : data(16) {}
456 
457  auto size() const { return sz; }
458  auto empty() const { return size() == 0; }
459 
460  auto rehash(size_t count)
461  {
462  if (empty()) {
463  size_t capacity = 16;
464  while (capacity <= count)
465  capacity <<= 1;
466  data.resize(capacity);
467  max_load_factor_capacity =
468  std::ceil(capacity * max_load_fact);
469  return;
470  }
471  if (count < size() / max_load_fact)
472  count = size() / max_load_fact;
473  auto n = Hash_Multiset();
474  n.rehash(count);
475  for (auto& b : data) {
476  for (auto& x : b) {
477  n.insert(x);
478  }
479  }
480  data.swap(n.data);
481  sz = n.sz;
482  max_load_factor_capacity = n.max_load_factor_capacity;
483  }
484 
485  auto reserve(size_t count) -> void
486  {
487  rehash(std::ceil(count / max_load_fact));
488  }
489 
490  auto insert(const_reference value)
491  {
492  using namespace std;
493  if (sz == max_load_factor_capacity) {
494  reserve(sz + 1);
495  }
496  auto&& key = key_extract(value);
497  auto h = hash(key);
498  auto h_mod = h & (data.size() - 1);
499  auto& bucket = data[h_mod];
500  if (bucket.size() == 0 || bucket.size() == 1 ||
501  equal(key, key_extract(bucket.back()))) {
502  bucket.push_back(value);
503  ++sz;
504  return end(bucket) - 1;
505  }
506  auto last =
507  std::find_if(rbegin(bucket), rend(bucket), [&](auto& x) {
508  return equal(key, key_extract(x));
509  });
510  if (last != rend(bucket)) {
511  auto ret = bucket.insert(last.base(), value);
512  ++sz;
513  return ret;
514  }
515 
516  bucket.push_back(value);
517  ++sz;
518  return end(bucket) - 1;
519  }

521  auto emplace(Args&&... a)
522  {
523  return insert(value_type(std::forward<Args>(a)...));
524  }
