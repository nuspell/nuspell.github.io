427 class Hash_Multiset {
428  private:
429  using bucket_type = boost::container::small_vector<Value, 1>;
430  static constexpr float max_load_fact = 7.0 / 8.0;
431  std::vector<bucket_type> data;
432  size_t sz = 0;
433  size_t max_load_factor_capacity = 0;
434  KeyExtract key_extract = {};
435  Hash hash = {};
436  KeyEqual equal = {};
437 
438  public:
439  using key_type = Key;
440  using value_type = Value;
441  using size_type = std::size_t;
442  using difference_type = std::ptrdiff_t;
443  using hasher = Hash;
444  using reference = value_type&;
445  using const_reference = const value_type&;
446  using pointer = typename bucket_type::pointer;
447  using const_pointer = typename bucket_type::const_pointer;
448  using local_iterator = typename bucket_type::iterator;
449  using local_const_iterator = typename bucket_type::const_iterator;

451  Hash_Multiset() : data(16) {}
452 
453  auto size() const { return sz; }
454  auto empty() const { return size() == 0; }
455 
456  auto rehash(size_t count)
457  {
458  if (empty()) {
459  size_t capacity = 16;
460  while (capacity <= count)
461  capacity <<= 1;
462  data.resize(capacity);
463  max_load_factor_capacity =
464  std::ceil(capacity * max_load_fact);
465  return;
466  }
467  if (count < size() / max_load_fact)
468  count = size() / max_load_fact;
469  auto n = Hash_Multiset();
470  n.rehash(count);
471  for (auto& b : data) {
472  for (auto& x : b) {
473  n.insert(x);
474  }
475  }
476  data.swap(n.data);
477  sz = n.sz;
478  max_load_factor_capacity = n.max_load_factor_capacity;
479  }
480 
481  auto reserve(size_t count) -> void
482  {
483  rehash(std::ceil(count / max_load_fact));
484  }
485 
486  auto insert(const_reference value)
487  {
488  using namespace std;
489  if (sz == max_load_factor_capacity) {
490  reserve(sz + 1);
491  }
492  auto&& key = key_extract(value);
493  auto h = hash(key);
494  auto h_mod = h & (data.size() - 1);
495  auto& bucket = data[h_mod];
496  if (bucket.size() == 0 || bucket.size() == 1 ||
497  equal(key, key_extract(bucket.back()))) {
498  bucket.push_back(value);
499  ++sz;
500  return end(bucket) - 1;
501  }
502  auto last =
503  std::find_if(rbegin(bucket), rend(bucket), [&](auto& x) {
504  return equal(key, key_extract(x));
505  });
506  if (last != rend(bucket)) {
507  auto ret = bucket.insert(last.base(), value);
508  ++sz;
509  return ret;
510  }
511 
512  bucket.push_back(value);
513  ++sz;
514  return end(bucket) - 1;
515  }
