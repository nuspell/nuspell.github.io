84 class Locale_Input {
85  std::locale external_locale;
86  auto static cvt_for_byte_dict(const std::string& in,
87  const std::locale& inloc,
88  const std::locale& dicloc) -> std::string;

91 
92  public:
93  auto imbue(const std::locale& loc) { external_locale = loc; }
94  // getloc() calls in iostreams return by value doing a ref-counted copy
95  // of pointer. Doing that we pay little extra than just returning a
96  // const reference. Here we fix that and return a const reference.
97  auto& getloc() const { return external_locale; }
98  auto cvt_for_byte_dict(const std::string& in,
99  const std::locale& dic_loc) const -> std::string
100  {
101  return cvt_for_byte_dict(in, getloc(), dic_loc);
102  }
103  auto cvt_for_u8_dict(const std::string& in) const -> std::wstring
104  {
105  return cvt_for_u8_dict(in, getloc());
106  }
107 };
